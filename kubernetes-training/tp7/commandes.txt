kubectl apply -f https://raw.githubusercontent.com/kubernetes/dashboard/v2.4.0/aio/deploy/recommended.yaml

kubectl apply -f dashboard-admin.yml 
kubectl apply -f dashboard-read-only.yml




#Creation des token
kubectl -n kubernetes-dashboard describe secret $(kubectl -n kubernetes-dashboard get secret | grep dashboard-admin | awk '{print $1}')

#
kubectl -n kubernetes-dashboard describe secret $(kubectl -n kubernetes-dashboard get secret | grep dashboard-read-only | awk '{print $1}')

#Une fois les tokens creer 

#suivre les etapes suivantes:

- kubectl proxy 
- Accédez au lien suivant dans votre navigateur pour ouvrir le Dashboard officiel : http://localhost:8001/api/v1/namespaces/kubernetes-dashboard/services/https:kubernetes-dashboard:/proxy/
- Sélectionnez l'option "Token" lors de la connexion au Dashboard.
- Utilisez le token du serviceaccount "dashboard-admin" ou "dashboard-read-only" pour vous connecter.
